import java.util.*;

public class AutokeyCipher {

    // Encrypt plaintext using Autokey cipher
    public static String encrypt(String plaintext, String key) {
        plaintext = plaintext.toUpperCase().replaceAll("[^A-Z]", "");
        key = key.toUpperCase().replaceAll("[^A-Z]", "");

        StringBuilder fullKey = new StringBuilder(key);
        fullKey.append(plaintext); // Autokey: append plaintext to key

        StringBuilder ciphertext = new StringBuilder();
        for (int i = 0; i < plaintext.length(); i++) {
            int p = plaintext.charAt(i) - 'A';
            int k = fullKey.charAt(i) - 'A';
            int c = (p + k) % 26;
            ciphertext.append((char) ('A' + c));
        }

        return ciphertext.toString();
    }

    // Decrypt ciphertext using Autokey cipher
    public static String decrypt(String ciphertext, String key) {
        ciphertext = ciphertext.toUpperCase().replaceAll("[^A-Z]", "");
        key = key.toUpperCase().replaceAll("[^A-Z]", "");

        StringBuilder plaintext = new StringBuilder();
        StringBuilder fullKey = new StringBuilder(key);

        for (int i = 0; i < ciphertext.length(); i++) {
            int c = ciphertext.charAt(i) - 'A';
            int k = fullKey.charAt(i) - 'A';
            int p = (c - k + 26) % 26;
            plaintext.append((char) ('A' + p));
            fullKey.append((char) ('A' + p)); // extend key with decrypted letter
        }

        return plaintext.toString();
    }

    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);

        System.out.print("Enter key: ");
        String key = sc.nextLine();

        System.out.print("Enter plaintext: ");
        String plaintext = sc.nextLine();

        String encrypted = encrypt(plaintext, key);
        System.out.println("Encrypted: " + encrypted);

        String decrypted = decrypt(encrypted, key);
        System.out.println("Decrypted: " + decrypted);
    }
}
